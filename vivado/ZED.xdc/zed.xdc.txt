## 100 MHz clock - ZedBoard
set_property PACKAGE_PIN Y9 [get_ports clk_100MHz]
set_property IOSTANDARD LVCMOS33 [get_ports clk_100MHz]
create_clock -period 10.000 -name sys_clk -waveform {0 5} [get_ports clk_100MHz]

## Reset Button (BTNC)
set_property PACKAGE_PIN U18 [get_ports reset_rtl]
set_property IOSTANDARD LVCMOS33 [get_ports reset_rtl]
